    <script>
        let vendas=JSON.parse(localStorage.getItem('vendas'))||[];
        let precos={gas12kg:85,gas45kg:220,agua20l:8.5,agua12l:6};

        function showTab(tab){document.querySelectorAll('.tab-content').forEach(t=>t.classList.remove('active'));document.querySelectorAll('.tab').forEach(t=>t.classList.remove('active'));document.getElementById(tab).classList.add('active');event.target.classList.add('active');if(tab=='historico')atualizarTabela();if(tab=='relatorio')atualizarRelatorios()}

        function calcularTotal(){const p=document.getElementById('produto').value,q=parseInt(document.getElementById('quantidade').value)||0,t=precos[p]*q;document.getElementById('total').textContent=`R$ ${t.toFixed(2).replace('.',',')}`}

        document.getElementById('produto').addEventListener('change',calcularTotal);document.getElementById('quantidade').addEventListener('input',calcularTotal);

        function registrarVenda(){const c=document.getElementById('cliente').value.trim();if(!c){alert('❌ Nome do cliente obrigatório!');return}const p=document.getElementById('produto'),produto=p.options[p.selectedIndex].text,q=parseInt(document.getElementById('quantidade').value),total=precos[p.value]*q,venda={id:Date.now(),data:new Date().toLocaleString('pt-BR'),cliente:c,endereco:document.getElementById('endereco').value,telefone:document.getElementById('telefone').value,produto,quantidade:q,valorUnitario:precos[p.value],total,status:'pendente'};vendas.unshift(venda);localStorage.setItem('vendas',JSON.stringify(vendas));document.getElementById('cliente').value='';document.getElementById('endereco').value='';document.getElementById('telefone').value='';document.getElementById('quantidade').value=1;document.getElementById('produto').selectedIndex=0;calcularTotal();alert('✅ Venda registrada!');atualizarTabela();atualizarRelatorios()}

        function atualizarTabela(){const tbody=document.querySelector('#tabelaVendas tbody'),busca=document.getElementById('busca').value.toLowerCase();tbody.innerHTML='';vendas.filter(v=>v.cliente.toLowerCase().includes(busca)||v.produto.toLowerCase().includes(busca)).forEach(v=>{const row=tbody.insertRow();row.innerHTML=`<td>${v.data}</td><td>${v.cliente}</td><td>${v.produto}</td><td>${v.quantidade}</td><td>R$ ${v.total.toFixed(2).replace('.',',')}</td><td><span class="status ${v.status}">${v.status=='pago'?'✅ Pago':'⏳ Pendente'}</span></td><td><button class="btn ${v.status=='pago'?'':'btn-success'}" onclick="pagar(${v.id})" style="padding:6px 12px;font-size:12px">${v.status=='pago'?'✅':'💰'}</button><button class="btn btn-danger" onclick="excluir(${v.id})" style="padding:6px 12px;font-size:12px">🗑️</button></td>`})}

        function pagar(id){const v=vendas.find(v=>v.id===id);if(v){v.status='pago';localStorage.setItem('vendas',JSON.stringify(vendas));atualizarTabela();atualizarRelatorios();alert('✅ Pago!')}}

        function excluir(id){if(confirm('Excluir venda?')){vendas=vendas.filter(v=>v.id!==id);localStorage.setItem('vendas',JSON.stringify(vendas));atualizarTabela();atualizarRelatorios()}}

        function atualizarRelatorios(){const hoje=new Date().toDateString(),agora=new Date(),vh=vendas.filter(v=>new Date(v.data).toDateString()===hoje),vm=vendas.filter(v=>{const dv=new Date(v.data);return dv.getMonth()===agora.getMonth()&&dv.getFullYear()===agora.getFullYear()}),tg=vendas.reduce((s,v)=>s+v.total,0),th=vh.reduce((s,v)=>s+v.total,0),tm=vm.reduce((s,v)=>s+v.total,0);document.getElementById('totalHoje').textContent=`R$ ${th.toFixed(2).replace('.',',')}`;document.getElementById('totalMes').textContent=`R$ ${tm.toFixed(2).replace('.',',')}`;document.getElementById('totalGeral').textContent=`R$ ${tg.toFixed(2).replace('.',',')}`;document.getElementById('vendasHoje').textContent=`${vh.length} vendas`;document.getElementById('vendasMes').textContent=`${vm.length} vendas`;document.getElementById('vendasTotal').textContent=`${vendas.length} vendas`;const prod={};vendas.forEach(v=>{if(!prod[v.produto]){prod[v.produto]={qtd:0,total:0}}prod[v.produto].qtd+=v.quantidade;prod[v.produto].total+=v.total});const top=Object.entries(prod).sort(([,a],[,b])=>b.total-a.total).slice(0,5),tbody=document.getElementById('topProdutos');tbody.innerHTML='';top.forEach(([p,d])=>{const row=tbody.insertRow();row.innerHTML=`<td>${p}</td><td>${d.qtd}</td><td>R$ ${d.total.toFixed(2).replace('.',',')}</td>`})}

        document.getElementById('busca').addEventListener('input',atualizarTabela);calcularTotal();atualizarTabela();atualizarRelatorios();
    </script>
